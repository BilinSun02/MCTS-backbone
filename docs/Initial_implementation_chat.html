class MCTSNode(Generic[EdgeT]):
    """
    Minimal node per requirement:
      - pointer to parent
      - dict of edges to children: {edge_value -> child_node}

    Task-specific information (state, player to move, etc.) should live in a subclass.
    """

    __slots__ = ("parent", "children")

    def __init__(self, parent: Optional["MCTSNode[EdgeT]"] = None) -> None:
        self.parent: Optional["MCTSNode[EdgeT]"] = parent
        self.children: Dict[EdgeT, "MCTSNode[EdgeT]"] = {}

    # --- Infrastructure helpers (useful in expansion code) ---

    def attach_child(self, edge: EdgeT, child: "MCTSNode[EdgeT]") -> "MCTSNode[EdgeT]":
        """
        Attach an existing child node under the given edge.
        Sets child's parent and writes into children dict.
        """
        child.parent = self
        self.children[edge] = child
        return child

    def get_child(self, edge: EdgeT) -> Optional["MCTSNode[EdgeT]"]:
        return self.children.get(edge)

    def iter_children(self) -> Iterable[Tuple[EdgeT, "MCTSNode[EdgeT]"]]:
        return self.children.items()

@runtime_checkable
class MCTSProblem(Protocol[NodeT, EdgeT]):
    """
    Abstract environment/game interface for MCTS.

    Key design choice:
      - Expansion is *delegated* to the problem via `expand_one`.
      - MCTS provides helper `mcts.create_child(...)` to facilitate common actions.
    """

    def is_terminal(self, node: NodeT) -> bool:
        """
        True if node is terminal (no further actions / game ended).
        """
        ...

    def expand_one(self, node: NodeT, mcts: "MCTS[NodeT, EdgeT]") -> Optional[Tuple[EdgeT, NodeT]]:
        """
        Expand EXACTLY ONE new child from `node` if possible and return (edge, child).

        If `node` is fully expanded (or non-expandable), return None.

        Typical implementation:
          - compute legal actions
          - pick one unexpanded action
          - create/attach child using mcts.create_child(node, edge, child=..., or via node_factory)
          - return (edge, child)
        """
        ...

    def rollout(self, node: NodeT, rng: random.Random) -> float:
        """
        Simulate from `node` to obtain a scalar reward (higher is better).
        This can be:
          - random playout to terminal
          - heuristic evaluation at depth limit
          - direct terminal utility if node is terminal

        Convention: reward is from the perspective you want the tree to optimize at the ROOT.
        """
        ...

    def backprop_transform(self, node: NodeT, reward: float, depth_from_leaf: int) -> float:
        """
        Optional hook to transform reward during backprop.
        Default behavior for many single-agent problems: identity.

        For alternating-player zero-sum games you might do:
          return -reward if depth_from_leaf is odd else reward
        """
        return reward  # type: ignore[return-value]

class MCTS(Generic[NodeT, EdgeT]):
    """
    Generic MCTS engine:
      - stats stored externally (WeakKeyDictionary keyed by node identity)
      - selection uses UCB1 by default; can be replaced with any scoring function
      - expansion delegated to problem.expand_one(...)
    """

    def __init__(
        self,
        root: NodeT,
        *,
        exploration_constant: float = math.sqrt(2.0),
        score_fn: Optional[ScoreFn[NodeT, EdgeT]] = None,
        node_factory: Optional[Callable[..., NodeT]] = None,
        rng: Optional[random.Random] = None,
    ) -> None:
        self.root: NodeT = root
        self.c: float = exploration_constant
        self.rng: random.Random = rng if rng is not None else random.Random()

        # Optional node factory used by create_child(...) when caller doesn't pass child explicitly.
        # It MAY accept parent=...; if not, we'll set parent afterwards.
        self.node_factory: Optional[Callable[..., NodeT]] = node_factory

        # Externalized statistics (keeps base Node minimal).
        self._visits: "WeakKeyDictionary[NodeT, int]" = WeakKeyDictionary()
        self._value_sum: "WeakKeyDictionary[NodeT, float]" = WeakKeyDictionary()

        # Selection scoring:
        self.score_fn: ScoreFn[NodeT, EdgeT] = score_fn if score_fn is not None else self._ucb1_score

    # ---------- Stats API ----------

    def visits(self, node: NodeT) -> int:
        return self._visits.get(node, 0)

    def value_sum(self, node: NodeT) -> float:
        return self._value_sum.get(node, 0.0)

    def value_mean(self, node: NodeT) -> float:
        n = self.visits(node)
        return (self.value_sum(node) / n) if n > 0 else 0.0

    def stats(self, node: NodeT) -> NodeStats:
        return NodeStats(visits=self.visits(node), value_sum=self.value_sum(node))

    # ---------- Child creation helper (infrastructure for expansion) ----------

    def create_child(
        self,
        parent: NodeT,
        edge: EdgeT,
        *,
        child: Optional[NodeT] = None,
        factory_kwargs: Optional[Mapping[str, Any]] = None,
    ) -> NodeT:
        """
        Convenience function to create (if needed) and attach a new child under `parent` via `edge`.

        Expansion implementations typically call:
          child = mcts.create_child(node, action, child=MyNode(...))  # explicit
        or:
          child = mcts.create_child(node, action, factory_kwargs={...})  # via node_factory
        """
        if child is None:
            if self.node_factory is None:
                raise ValueError("No child provided and no node_factory configured.")
            kwargs = dict(factory_kwargs or {})
            # Try factory(parent=...) first; fall back to no-parent signature.
            try:
                child = self.node_factory(parent=parent, **kwargs)  # type: ignore[misc]
            except TypeError:
                child = self.node_factory(**kwargs)  # type: ignore[misc]
                child.parent = parent

        # Attach (also sets child's parent)
        parent.children[edge] = child
        child.parent = parent
        return child

    # ---------- Selection (default UCB1, but pluggable) ----------

    def _ucb1_score(self, parent: NodeT, edge: EdgeT, child: NodeT) -> float:
        """
        Vanilla UCB1 score:
          Q(child) + c * sqrt( ln(N(parent)) / N(child) )

        Unvisited children get +inf to ensure exploration.
        """
        n_child = self.visits(child)
        if n_child == 0:
            return float("inf")

        n_parent = self.visits(parent)
        # guard: if parent has no visits yet, treat ln(1)=0
        log_parent = math.log(max(1, n_parent))

        exploit = self.value_mean(child)
        explore = self.c * math.sqrt(log_parent / n_child)
        return exploit + explore

    def select_child(self, node: NodeT) -> Tuple[EdgeT, NodeT]:
        """
        Pick the best child under current selection policy (score_fn).
        """
        if not node.children:
            raise ValueError("select_child called on a node with no children.")

        best_edge: Optional[EdgeT] = None
        best_child: Optional[NodeT] = None
        best_score = -float("inf")

        for edge, child in node.children.items():
            score = self.score_fn(self, node, edge, child)
            if score > best_score:
                best_score = score
                best_edge = edge
                best_child = child

        # node.children non-empty => best_* must be set
        return best_edge, best_child  # type: ignore[return-value]

    # ---------- Core MCTS loop ----------

    def run(self, problem: MCTSProblem[NodeT, EdgeT], *, iterations: int) -> None:
        """
        Run a fixed number of MCTS iterations from the current root.
        """
        for _ in range(iterations):
            self._iterate(problem)

    def _iterate(self, problem: MCTSProblem[NodeT, EdgeT]) -> None:
        """
        One MCTS iteration:
          1) Selection + Expansion (tree policy)
          2) Rollout
          3) Backprop
        """
        node = self.root
        path: List[NodeT] = [node]

        # ---- Tree policy: alternate between expanding one child or selecting down ----
        while not problem.is_terminal(node):
            expanded = problem.expand_one(node, self)
            if expanded is not None:
                _, child = expanded
                node = child
                path.append(node)
                break  # rollout from the newly expanded node

            # no expansion possible => select among existing children
            if not node.children:
                break  # dead-end leaf
            _, node = self.select_child(node)
            path.append(node)

        # ---- Rollout / Evaluation ----
        reward = problem.rollout(node, self.rng)

        # ---- Backpropagation ----
        self._backprop(problem, path, reward)

    def _backprop(self, problem: MCTSProblem[NodeT, EdgeT], path: Sequence[NodeT], reward: float) -> None:
        """
        Update external stats along the path. Allows per-depth reward transforms.
        """
        # depth_from_leaf: 0 at leaf, 1 at parent, ...
        for depth_from_leaf, node in enumerate(reversed(path)):
            r = problem.backprop_transform(node, reward, depth_from_leaf)

            self._visits[node] = self.visits(node) + 1
            self._value_sum[node] = self.value_sum(node) + float(r)

    # ---------- Action recommendation ----------

    def best_edge(
        self,
        node: Optional[NodeT] = None,
        *,
        by: str = "visits",
    ) -> EdgeT:
        """
        Recommend an action from `node` (default: root).

        by="visits": pick edge whose child has max visits (robust default).
        by="value":  pick edge whose child has max mean value.
        """
        node = self.root if node is None else node
        if not node.children:
            raise ValueError("No children available to choose from.")

        if by not in ("visits", "value"):
            raise ValueError('by must be "visits" or "value".')

        best_edge: Optional[EdgeT] = None
        best_metric = -float("inf")

        for edge, child in node.children.items():
            metric = self.visits(child) if by == "visits" else self.value_mean(child)
            if metric > best_metric:
                best_metric = metric
                best_edge = edge

        return best_edge  # type: ignore[return-value]

class MCTS(Generic[NodeT, EdgeT]):
    """
    Generic, abstract MCTS engine with reentrancy/parallelism support.

    Concurrency model:
      - node structure (node.children, node.parent) is protected by per-node RLocks.
      - stats (visits/value_sum/in_flight) are protected by per-node locks (in _Stats.mu).
      - tables mapping node -> lock/stats are WeakKeyDictionaries; their get-or-create
        is protected by a global table lock.

    Thread-safety rules (do not violate these in downstream specializations):
      1) NEVER iterate node.children without holding the node’s struct lock, or using
         the snapshot helpers; otherwise you risk “dict changed size during iteration”. :contentReference[oaicite:5]{index=5}
      2) ONLY mutate node.children / node.parent via MCTS.try_create_child/create_child;
         do not write node.children[...] directly in your problem code. :contentReference[oaicite:6]{index=6}
      3) NEVER do stats read-modify-write without holding _Stats.mu, or increments will drop. :contentReference[oaicite:7]{index=7}
      4) Avoid deadlocks: do not hold more than one node-structure lock at a time. :contentReference[oaicite:8]{index=8}

    Practical note:
      Python threads share the GIL. You’ll still see speedups when rollouts/expansion
      do I/O, call external processes, or run heavy C/CUDA kernels that release the GIL.
      For pure Python CPU-bound rollouts, consider multiprocessing (not implemented here).
    """

    def __init__(
        self,
        root: NodeT,
        *,
        exploration_constant: float = math.sqrt(2.0),
        score_fn: Optional[ScoreFn[NodeT, EdgeT]] = None,
        node_factory: Optional[Callable[..., NodeT]] = None,
        seed: Optional[int] = None,
        debug: bool = False,
    ) -> None:
        self.root: NodeT = root
        self.c: float = exploration_constant
        self.node_factory: Optional[Callable[..., NodeT]] = node_factory
        self.debug: bool = debug

        # Global mutex guarding "get-or-create" on our side tables. :contentReference[oaicite:9]{index=9}
        self._tables_mu = threading.Lock()

        # node -> per-node structure lock (protects children/parent publication). :contentReference[oaicite:10]{index=10}
        self._struct_locks: "WeakKeyDictionary[NodeT, threading.RLock]" = WeakKeyDictionary()

        # node -> stats record (protected internally by _Stats.mu).
        self._stats: "WeakKeyDictionary[NodeT, _Stats]" = WeakKeyDictionary()

        # Score function calling convention must be consistent. :contentReference[oaicite:11]{index=11}
        self.score_fn: ScoreFn[NodeT, EdgeT] = score_fn or (lambda m, p, e, c: m._ucb1_score(p, e, c))

        # Thread-local RNG: do not share a single random.Random across threads. :contentReference[oaicite:12]{index=12}
        self._rng_local = threading.local()
        self._base_seed = seed if seed is not None else random.randrange(1 << 63)

    # ----------------------------
    # Internal: table accessors
    # ----------------------------

    def _struct_mu(self, node: NodeT) -> threading.RLock:
        # WeakKeyDictionary is not thread-safe; guard get-or-create with a global lock. :contentReference[oaicite:13]{index=13}
        with self._tables_mu:
            mu = self._struct_locks.get(node)
            if mu is None:
                mu = threading.RLock()
                self._struct_locks[node] = mu
            return mu

    def _get_stats(self, node: NodeT) -> _Stats:
        # Guard get-or-create with global lock as recommended. :contentReference[oaicite:14]{index=14}
        with self._tables_mu:
            st = self._stats.get(node)
            if st is None:
                st = _Stats()
                self._stats[node] = st
            return st

    def _thread_rng(self) -> random.Random:
        """
        Per-thread RNG. Reentrant by construction (thread-local). :contentReference[oaicite:15]{index=15}
        """
        rng = getattr(self._rng_local, "rng", None)
        if rng is None:
            # Mix in thread identity so multiple threads diverge deterministically given base_seed.
            tid = threading.get_ident()
            self._rng_local.rng = random.Random(self._base_seed ^ (tid * 0x9E3779B97F4A7C15))
            rng = self._rng_local.rng
        return rng

    # ----------------------------
    # Stats API (thread-safe snapshots)
    # ----------------------------

    def stats_snapshot(self, node: NodeT) -> NodeStats:
        st = self._get_stats(node)
        with st.mu:
            return NodeStats(visits=st.visits, value_sum=st.value_sum, in_flight=st.in_flight)

    def visits(self, node: NodeT, *, effective: bool = False) -> int:
        s = self.stats_snapshot(node)
        return s.effective_visits if effective else s.visits

    def value_mean(self, node: NodeT) -> float:
        return self.stats_snapshot(node).value_mean

    # ----------------------------
    # Children/structure helpers (thread-safe)
    # ----------------------------

    def children_items_snapshot(self, node: NodeT) -> List[Tuple[EdgeT, NodeT]]:
        """
        Snapshot (edge, child) list under struct lock.

        Never iterate node.children directly under concurrency. :contentReference[oaicite:16]{index=16}
        """
        with self._struct_mu(node):
            # Copy to avoid "dict changed size during iteration" if expansion runs concurrently.
            return list(node.children.items())

    def children_keys_snapshot(self, node: NodeT) -> set[EdgeT]:
        """
        Snapshot expanded edge keys under struct lock.
        Used by parallel-safe expansion patterns. :contentReference[oaicite:17]{index=17}
        """
        with self._struct_mu(node):
            return set(node.children.keys())

    def try_create_child(
        self,
        parent: NodeT,
        edge: EdgeT,
        *,
        child: Optional[NodeT] = None,
        factory_kwargs: Optional[Mapping[str, Any]] = None,
    ) -> Tuple[NodeT, bool]:
        """
        Atomically "check-and-attach" a child under (parent, edge).

        Returns: (child_node, created_bool)

        Publication rules:
          - Must set child.parent BEFORE publishing parent.children[edge] to prevent readers
            observing a reachable child with a stale/None parent pointer. :contentReference[oaicite:18]{index=18}

        Duplicate edge policy:
          - Under parallel expand, benign races are expected.
          - If another thread already created the child for `edge`, return the existing child
            rather than raising (raising would create heisenbugs at scale). :contentReference[oaicite:19]{index=19}

        Performance note:
          - If you pass `child=...`, you may sometimes “waste” child construction when you lose
            the race. This is usually acceptable and helps keep locks held briefly.
        """
        # Construct candidate child outside the struct lock if needed (reduces lock hold time).
        if child is None:
            if self.node_factory is None:
                raise ValueError("No child provided and no node_factory configured.")
            kwargs = dict(factory_kwargs or {})
            try:
                child = self.node_factory(parent=parent, **kwargs)  # type: ignore[misc]
            except TypeError:
                child = self.node_factory(**kwargs)  # type: ignore[misc]

        with self._struct_mu(parent):
            existing = parent.children.get(edge)
            if existing is not None:
                if self.debug:
                    assert existing.parent is parent, "Child published under parent, but child.parent != parent"
                return existing, False

            # Publish in the correct order (parent pointer first), under lock. :contentReference[oaicite:20]{index=20}
            child.parent = parent
            parent.children[edge] = child

            if self.debug:
                assert parent.children[edge].parent is parent

            return child, True

        self,
        parent: NodeT,
        edge: EdgeT,
        *,
        child: Optional[NodeT] = None,
        factory_kwargs: Optional[Mapping[str, Any]] = None,
    ) -> NodeT:
        """
        Convenience wrapper that *ensures* parent has a child under `edge`, returning it.
        Under concurrency, this may return an existing child if another thread won the race.
        """
        node, _created = self.try_create_child(parent, edge, child=child, factory_kwargs=factory_kwargs)
        return node

        """
        Increment in_flight for `node` to reduce selection collisions between threads.
        Recommended by the attachment. :contentReference[oaicite:21]{index=21}
        """
        st = self._get_stats(node)
        with st.mu:

    def _ucb1_score(self, parent: NodeT, edge: EdgeT, child: NodeT) -> float:
        """
        UCB1 with virtual visits:
          Q(child) + c * sqrt( ln(N(parent)) / N(child) )

        Notes:
          - Q(child) uses real mean value (value_sum / visits).
          - N(*) uses effective visits (visits + in_flight) to discourage collisions.
        """
        parent_s = self.stats_snapshot(parent)
        child_s = self.stats_snapshot(child)

        n_child = child_s.effective_visits
        if n_child == 0:
            return float("inf")

        n_parent = max(1, parent_s.effective_visits)
        exploit = child_s.value_mean
        explore = self.c * math.sqrt(math.log(n_parent) / n_child)
        return exploit + explore

    def select_child(self, node: NodeT) -> Tuple[EdgeT, NodeT]:
        """
        Pick best child under current selection policy.
        Safe under concurrency because we snapshot children under struct lock. :contentReference[oaicite:22]{index=22}
        """
        items = self.children_items_snapshot(node)
        if not items:
            raise ValueError("select_child called on a node with no children.")

        best_score = -float("inf")
        best: List[Tuple[EdgeT, NodeT]] = []

        # IMPORTANT: score_fn runs WITHOUT holding the struct lock.
        # Do not write score_fn implementations that iterate node.children directly.
        for edge, child in items:
            score = self.score_fn(self, node, edge, child)
            if score > best_score:
                best_score = score
                best = [(edge, child)]
            elif score == best_score:
                best.append((edge, child))

        # Tie-break deterministically per-thread.
        rng = self._thread_rng()
        return rng.choice(best)

    # ----------------------------
    # Core MCTS iteration (parallel-safe)
    # ----------------------------

    def _iterate(self, problem: MCTSProblem[NodeT, EdgeT]) -> None:
        """
        One MCTS iteration:
          1) Selection / Expansion (tree policy)
          2) Rollout
          3) Backprop

        Concurrency details:
          - We add a virtual visit to each node on the traversed path (including root),
            then "resolve" it during backprop by decrementing in_flight and applying real updates.
            This matches the attachment’s recommended in_flight workflow. :contentReference[oaicite:23]{index=23}
          - We do NOT hold struct locks while calling user code (problem.expand_one / rollout),
            because those can be expensive and would destroy scalability. :contentReference[oaicite:24]{index=24}
        """
        node = self.root
        path: List[NodeT] = [node]

        # Virtual-visit root immediately to mark this iteration “in-flight”.
        self._virtual_visit(node)

        try:
            # ---- Tree policy ----
            while not problem.is_terminal(node):
                # Let the problem attempt to expand one child.
                # Expansion must use try_create_child/create_child, not direct dict writes. :contentReference[oaicite:25]{index=25}
                expanded = problem.expand_one(node, self)
                if expanded is not None:
                    _edge, child = expanded
                    node = child
                    path.append(node)
                    self._virtual_visit(node)
                    break  # rollout from the newly expanded node

                # No expansion possible => select among existing children.
                items = self.children_items_snapshot(node)
                if not items:
                    break  # dead-end leaf

                _edge, child = self.select_child(node)
                node = child
                path.append(node)
                self._virtual_visit(node)

            # ---- Rollout / Evaluation ----
            reward = problem.rollout(node, self._thread_rng())

            # ---- Backprop: resolve virtual visits and apply real stats updates ----
            self._backprop(problem, path, reward)

        except BaseException:
            # On exceptions, we must still resolve virtual visits, or in_flight will leak forever.
            self._resolve_virtual_visits_only(path)
            raise

    def _resolve_virtual_visits_only(self, path: Sequence[NodeT]) -> None:
        """
        Best-effort cleanup for in_flight if iteration fails mid-flight.
        Only decrements in_flight (no real stats updates).
        """
        for node in reversed(path):
            st = self._get_stats(node)
            with st.mu:
                st.in_flight -= 1
                if self.debug:
                    assert st.in_flight >= 0, "in_flight went negative (virtual-visit imbalance)"

    def _backprop(self, problem: MCTSProblem[NodeT, EdgeT], path: Sequence[NodeT], reward: float) -> None:
        """
        For each node on path (leaf->root):
          - decrement in_flight
          - increment visits
          - add transformed reward to value_sum

        This is the “race-free backprop” fix from the attachment. :contentReference[oaicite:26]{index=26}
        """
        for depth_from_leaf, node in enumerate(reversed(path)):
            r = float(problem.backprop_transform(node, reward, depth_from_leaf))
            st = self._get_stats(node)
            with st.mu:
                st.in_flight -= 1
                st.visits += 1
                st.value_sum += r
                if self.debug:
                    assert st.in_flight >= 0, "in_flight went negative (virtual-visit imbalance)"

    # ----------------------------
    # Public API: run single-threaded or parallel
    # ----------------------------

    def run(self, problem: MCTSProblem[NodeT, EdgeT], *, iterations: int) -> None:
        for _ in range(iterations):
            self._iterate(problem)

    def run_parallel(
        self,
        problem: MCTSProblem[NodeT, EdgeT],
        *,
        iterations: int,
        workers: int,
        chunk_size: int = 32,
    ) -> None:
        """
        Parallel runner using threads.

        Root management:
          - This implementation assumes “stop-the-world between moves”: do not change self.root
            while workers are running, as recommended in the attachment. :contentReference[oaicite:27]{index=27}
          - Reading best_edge() concurrently is OK (it snapshots children under lock and reads stats).

        Budget model:
          - workers pull chunks of iterations from a shared counter (reduces contention).
        """
        if iterations <= 0 or workers <= 0:
            return

        budget_mu = threading.Lock()
        remaining = {"n": iterations}  # mutable box

        stop = threading.Event()
        exc_mu = threading.Lock()
        first_exc: List[BaseException] = []

        def acquire_chunk() -> int:
            with budget_mu:
                if remaining["n"] <= 0:
                    return 0
                k = min(chunk_size, remaining["n"])
                remaining["n"] -= k
                return k

        def worker_main() -> None:
            try:
                while not stop.is_set():
                    k = acquire_chunk()
                    if k == 0:
                        return
                    for _ in range(k):
                        self._iterate(problem)
            except BaseException as e:
                stop.set()
                with exc_mu:
                    if not first_exc:
                        first_exc.append(e)

        threads = [threading.Thread(target=worker_main, name=f"mcts-worker-{i}") for i in range(workers)]
        for t in threads:
            t.start()
        for t in threads:
            t.join()

        if first_exc:
            # Re-raise the first exception from any worker (with context).
            raise RuntimeError("run_parallel: worker raised an exception") from first_exc[0]

    # ----------------------------
    # Action recommendation (parallel-safe)
    # ----------------------------

    def best_edge(self, node: Optional[NodeT] = None, *, by: str = "visits") -> EdgeT:
        """
        Recommend an action from `node` (default: root), safe under concurrency.

        by="visits": pick child with max real visits (robust).
        by="value":  pick child with max mean value (can be noisy early).
        """
        node = self.root if node is None else node
        items = self.children_items_snapshot(node)
        if not items:
            raise ValueError("No children available to choose from.")

        if by not in ("visits", "value"):
            raise ValueError('by must be "visits" or "value".')

        best_metric = -float("inf")
        best: List[EdgeT] = []

        for edge, child in items:
            s = self.stats_snapshot(child)
            metric = s.visits if by == "visits" else s.value_mean
            if metric > best_metric:
                best_metric = metric
                best = [edge]
            elif metric == best_metric:
                best.append(edge)

        return self._thread_rng().choice(best)

Yes, a good interface is possible, and I would not leave this entirely as an ad hoc burden on downstream code.

The key point is that your “extra free information” is not just an optimization detail. It changes the natural unit of expansion from:

  • “expand one edge from this node”

to something closer to:

  • “materialize a discovered chain / subtree fragment rooted at this node”

If the backbone continues to pretend the primitive is only “one direct child,” then every specialization that wants to exploit deeper free structure will end up re-implementing the same subtle concurrency logic around publication, duplicate detection, partial commit, and rollback boundaries. That is exactly the sort of thing the backbone should centralize.

The design question

There are really three possible interface shapes.

1. Keep the current interface: expand_one(node, mcts) -> Optional[(edge, child)]

and let the user create extra descendants on the side.

This is the simplest surface API, but I would treat it as insufficient as the main abstraction.

Why it is weak:

  • It obscures the true semantic event. Expansion is no longer really “one child.”

  • It forces downstream code to perform multi-node publication itself.

  • It makes parallel-safety conventions much harder to standardize.

  • It becomes unclear what counts as the “expanded result” of one tree-policy step if a chain of descendants was also created.

This interface is still useful as a lowest common denominator, but not as the primary one for your special case.

2. Replace it with a more general interface:

something like “expand and publish a fragment”

Conceptually:

  • the application computes some local expansion payload at a node

  • that payload may describe:

    • one direct child

    • several direct children

    • one direct child plus nested descendants beneath it

    • even several rooted chains, if your domain naturally gives that

Then the MCTS backbone owns the publication protocol.

This is the direction I would recommend.

3. Support both:

a simple one-child API and a richer fragment API, with the one-child API implemented as a degenerate fragment.

This is probably the best engineering choice.

You preserve a simple path for ordinary domains, while making the “multi-child / nested-child expansion” case first-class and safe.

What the richer abstraction should look like

I would separate expansion into two phases:

Phase A: discovery

User code computes an expansion fragment off-lock.

This fragment is a pure description of what could be added, not yet committed into the shared tree.

That fragment might contain:

  • the anchor node being expanded

  • one or more root outgoing edges from that anchor

  • for each such edge, a child node descriptor

  • optionally nested descendants under those newly discovered children

  • optional metadata about which newly added node should be treated as the rollout start node for this iteration

The important thing is that this fragment is detached from the live tree while being constructed.

That gives you:

  • no long lock hold times

  • no partially visible shared mutations during expensive inference/generation

  • a clean boundary between “compute what I found” and “attempt to publish it”

Phase B: publication

The backbone takes the fragment and tries to merge it into the actual tree under the appropriate locks.

This is where correctness under parallelism is enforced.

Why a “fragment” abstraction is better than “create many”

“Create many” sounds operational, but the real issue is not quantity. It is atomicity semantics and conflict handling.

A fragment abstraction lets you define those explicitly.

For example, you can decide whether publication is:

A. root-edge atomic only

For each direct edge from the expanded node:

  • either that direct child gets attached, or it already exists

  • once that child is known, deeper descendants beneath it can be attached independently

This is usually the most practical option.

Why it is attractive:

  • contention is naturally centered at the parent being expanded

  • duplicate races mostly matter at the first unseen edge

  • once a child exists, nested publication can proceed relative to that child

  • you do not need a giant transaction spanning the whole discovered chain

B. whole-fragment atomic

Either the entire discovered nested structure is published or none of it is.

I would generally avoid this unless your semantics truly require it.

Why:

  • it implies multi-node transactional publication

  • it increases lock complexity

  • it creates deadlock pressure

  • it is usually unnecessary for MCTS correctness

MCTS generally tolerates partial wins in a race as long as the tree remains structurally consistent.

Recommended publication semantics

I would recommend this rule:

The backbone guarantees structural consistency and idempotent merge of discovered fragments, but not transactional all-or-nothing publication of the entire fragment.

Concretely:

  • each parent→edge→child attachment is individually linearizable

  • if another thread already published that edge, the backbone reuses the existing child

  • nested descendants are then merged relative to the canonical child that ended up in the tree

  • losing some nested publications to races is fine; they can be rediscovered or completed later

This gives you the right balance:

  • correctness

  • simplicity

  • reuse of free downstream information

  • no need for full transactions

What should the fragment contain?

At sketch level, I would make the application return a tree-shaped object, not live mutations.

Something like:

  • ExpansionFragment

    • anchor: the node being expanded

    • entries: collection of discovered root outgoing edges

    • each entry describes:

      • edge label

      • child node object or child construction spec

      • optional nested subfragment under that child

    • rollout_node_selector or a designated node within the fragment

There are two variants here.

Variant 1: fragment contains actual node instances

User constructs the child/grandchild/etc. node objects up front.

Pros:

  • simplest for user code

  • natural if the expensive expansion already computed the exact downstream states

Cons:

  • races may cause some constructed nodes to be discarded unused

  • can waste memory/work

Usually acceptable if expansion is expensive enough that reconstructing later would be worse.

Variant 2: fragment contains descriptors / factories

User returns enough information to lazily construct nodes only if publication succeeds.

Pros:

  • avoids waste under heavy contention

Cons:

  • more complicated interface

  • harder to keep generic

  • may force user to split logic awkwardly

I would start with Variant 1 unless contention is known to be severe.

How tree policy should interpret such an expansion

You need to decide what one MCTS iteration does after publishing a fragment.

There are two natural choices.

Choice 1: stop at the first newly materialized node and rollout from there

This is the closest analog to ordinary MCTS “expand one new child, then rollout.”

Even if a deeper chain was published too, the current iteration conceptually expanded only one frontier step.

Pros:

  • preserves standard MCTS feel

  • easiest to reason about statistically

Cons:

  • does not fully exploit the free deeper information during this particular iteration

Choice 2: allow the expansion result to nominate a deeper node as the rollout start

For example, if expansion naturally gives you a deterministic chain parent -> c1 -> c2 -> c3, you might rollout from c3.

This is very reasonable if those intermediate nodes were not stochastic choices but forced or logically implied transitions.

Pros:

  • fully leverages the free work

  • matches the computational reality

Cons:

  • changes the effective notion of what “one expansion” means

  • may subtly alter visit statistics distribution versus vanilla MCTS

I would support this explicitly in the interface rather than leave it implicit.

So the fragment publication result should probably return:

  • which nodes were newly attached

  • which canonical node should be considered the “selected leaf” for rollout/backprop in this iteration

What about visit accounting and virtual visits?

This is an important subtlety.

Suppose one iteration expands a fragment that includes multiple nested new nodes. Which of them get virtual visits / real visits on this iteration?

I would recommend:

  • The selection path gets virtual visits as usual.

  • After expansion, the iteration chooses a single rollout start node.

  • The backprop path is the ancestor chain from that rollout start node up to the root.

  • Nodes that were published in the fragment but are not on that ancestor chain get no visit update yet.

Why this is the clean rule:

  • visits represent simulated trajectories, not mere existence

  • publication of extra descendants should not by itself count as having traversed them unless this iteration actually uses them as the leaf path

  • otherwise statistics become inflated by “free construction” rather than search traffic

So the backbone should distinguish:

  • materialized nodes

  • visited by this simulation nodes

That distinction becomes important once expansion can create multiple nested descendants.

Parallel correctness issues specific to fragment publication

Here are the main ones the interface should make explicit.

1. Publication must be merge-like, not overwrite-like

If another thread already created some prefix of your discovered structure, you must merge into that canonical prefix rather than blindly attach your own duplicate nodes.

So publication should behave like:

  • “find or attach canonical child for edge”

  • then recurse / continue under that canonical child

2. No requirement that the fragment remain entirely novel

The fragment may be stale by the time it is published.

That is normal. The interface should explicitly say that publication is best-effort and idempotent.

3. User code should not directly mutate live descendants during discovery

The application should compute a detached fragment, not walk the live tree mutating grandchildren directly. Otherwise you lose the backbone’s ability to enforce locking discipline.

4. Avoid multi-lock transactional commit

Do not require the publication of a nested fragment to hold a whole chain of locks simultaneously. That is where deadlock complexity explodes.

Prefer:

  • attach or reuse one edge under one parent lock

  • release

  • continue at child

That may allow interleaving with other threads, which is fine.

A good interface shape

At a high level, I would evolve the protocol from:

  • expand_one(node, mcts)

toward something like:

  • discover_expansion(node, mcts) -> Optional[ExpansionFragment]

  • backbone method publish_fragment(fragment) -> PublicationResult

or equivalently a single problem hook:

  • expand(node, mcts) -> Optional[ExpansionPlan]

where the returned plan is declarative.

The key is that the user describes discovered structure, but the backbone owns:

  • canonicalization against races

  • lock discipline

  • duplicate-edge behavior

  • mapping from discovered nodes to actual attached nodes

  • choice of rollout start node semantics

Should the backbone also support “multiple direct children but no nested descendants”?

Yes. That should just be a trivial fragment with several root entries.

This is another reason the fragment abstraction is the right one: it cleanly subsumes:

  • one child

  • many siblings

  • one chain

  • several small rooted subtrees

My recommendation

I would not keep the old interface as the only interface.

I would do this instead:

  • Keep the one-child API as a compatibility/simple-case layer.

  • Introduce a first-class expansion fragment / expansion plan abstraction.

  • Make the backbone responsible for publishing/merging such fragments safely.

  • Define statistics so that only the chosen rollout path receives visit updates, not every newly materialized node.

  • Keep publication non-transactional but structurally linearizable per edge attachment.

That gives you:

  • efficiency from reusing free nested information

  • a clean generic abstraction

  • centralized parallel-safety logic

  • no need for downstream users to reinvent tricky concurrent subtree merge code

What I would explicitly document for downstream implementers

I would put these rules in the backbone docs/comments:

  • Expansion discovery may compute arbitrarily large detached fragments.

  • Discovery must not mutate the live tree.

  • Publication is merge-like and race-tolerant.

  • Another thread may already have materialized part or all of the fragment.

  • The backbone may reuse canonical existing nodes instead of the fragment’s node objects.

  • Materialization does not imply a visit.

  • Only the rollout-start ancestry is backpropagated for that iteration.

  • Implementers should treat fragment construction as speculative; some constructed nodes may be discarded if races are lost.

That is the conceptual interface boundary I would aim for.

Because in a concurrent setting, publishing a whole fragment atomically usually means you need something transaction-like across multiple shared objects.

In your setting, those shared objects are multiple nodes’ child dictionaries:

  • A.children

  • B.children

  • C.children

  • lock all relevant nodes

  • verify nothing conflicting happened

  • install all edges

  • then release locks

  • You may need to hold multiple locks at once

  • Different threads may discover overlapping fragments and lock nodes in different orders

  • Even without deadlock, lock hold times get longer

  • each individual edge insertion is correct

  • the tree is always structurally consistent

  • duplicate races are resolved cleanly

  • partial insertion is allowed

  • A -> B

and later:

  • A -> B -> C

and later:

  • A -> B -> C -> D

That is much easier to support, and it is generally sufficient.

During expansion at A, your domain logic gives enough information to construct this chain:

  • A --e1--> B

  • B --e2--> C

  • C --e3--> D

Now you have to decide what this iteration counts as having actually explored.

Case 1: rollout starts at D

Then the traversed path for this iteration is effectively:

  • root

  • ...

  • A

  • B

  • C

  • D

In that case, B, C, and D are on the ancestor chain of the rollout start, so they do get visit updates.

Case 2: rollout starts at B

This is the case I was referring to.

You still publish C and D because expansion gave them to you for free, but this iteration only conceptually advanced the search frontier to B.

Then the backprop path is:

  • root

  • ...

  • A

  • B

and not through C or D.

So:

  • C and D are now in the tree

  • but they get no visit update yet

That is exactly the situation I meant.

Another example: many siblings

Suppose expansion at A computes three direct children at once:

  • A --m1--> B1

  • A --m2--> B2

  • A --m3--> B3

but for this iteration you choose B2 as the rollout start.

Then:

  • B1, B2, B3 may all be published

  • but only B2 is on the simulation path for this iteration

So:

  • B2 gets the visit/backprop contribution

  • B1 and B3 do not

They exist, but they were not actually traversed by this simulation.

Why this rule exists

Because otherwise you would be conflating:

  • “this node was cheap to construct” with

  • “this node was actually sampled by MCTS”

Those are different.

Visits are supposed to represent something like:

  • how many simulations passed through this node

not:

  • how many times expansion happened to reveal this node as a side effect

If you gave visits to every newly materialized node, then a single expansion that happens to reveal a large subtree would artificially inflate statistics for nodes the search never actually chose.

When it comes up in practice

It comes up whenever your special expansion returns any of these:

  • one direct child plus a forced continuation chain

  • multiple direct children, but the iteration only rolls out from one

  • a partial subtree rooted at the expanded node

  • deterministic “free” descendant states produced during inference

In all such cases, some nodes may be:

  • inserted into the tree now

  • but not yet part of any simulation path

Those nodes should typically wait until a later iteration actually traverses them before receiving visit counts.

The core criterion

A node gets a visit update in a given iteration only if that node lies on the chosen path:

  • from root

  • to the node where rollout/evaluation is considered to happen

Any other nodes published opportunistically during the same expansion do not get that update yet.

The fragment may be stale by the time it is published.

And what does this mean?

It means:

by the time your thread finishes computing a proposed expansion fragment and tries to attach it to the shared MCTS tree, other threads may already have changed that part of the tree, so your proposal is no longer “fresh.”

Concrete example

Suppose thread 1 is expanding node A.

It computes, off to the side, a fragment saying:

  • add edge e1 from A to new child B

  • under B, add edge e2 to C

But while thread 1 is still doing that computation, thread 2 also reaches A and successfully publishes:

  • A --e1--> B'

Now thread 1 comes back and says “I want to publish my fragment.”

That fragment is now stale in the sense that it was computed against an older view of the tree:

  • when thread 1 started, A had no e1 child

  • when thread 1 finishes, A already does have one

So thread 1 cannot just assume its fragment is still exactly right as-is.

What should happen instead

Publication should treat the fragment as a proposal, not as an unquestionable fact.

So when thread 1 tries to publish:

  • it checks whether A.children[e1] already exists

  • if yes, it reuses that canonical existing child instead of insisting on its own B

  • then it can continue trying to merge the rest of its discovered structure under that canonical child

So “stale” does not mean “useless.” It just means:

  • the world moved on while you were computing

  • you need merge logic rather than blind insertion

Why this is normal

Because in a parallel MCTS, discovery and publication are separated in time:

  1. you inspect a node

  2. you do some possibly expensive computation

  3. only afterward do you acquire locks and try to commit

During step 2, other threads may have:

  • added children

  • added some of the same descendants

  • fully completed the expansion you were about to do

So staleness is not an error condition. It is the default expectation in concurrent search.

A simpler analogy

It is like drafting edits to a shared document:

  • you start from version 10

  • while you are typing, someone else edits the document to version 11

  • when you go to save, your edit is based on an older snapshot

Your edit is now “stale.” You do not necessarily discard it, but you have to merge it against the current version.

In one sentence

“The fragment may be stale by the time it is published” means that the fragment was computed from an earlier snapshot of the tree, and by commit time another thread may already have inserted overlapping structure, so publication must merge/reuse rather than assume the fragment is still novel.

But in case of such a "conflict", what's the correct behavior? If thread 1 conceived B --e2--> C, do we now create B' --e2--> C or do we just give up everything downstream to B?

Usually, neither extreme.

The right default is:

reuse the canonical node that actually won the race in the tree, and then continue trying to merge the downstream fragment under that canonical node.

So in your example, if thread 2 already published

  • A --e1--> B'

while thread 1 had computed

  • A --e1--> B

  • B --e2--> C

then the usual behavior should be:

  • do not publish your private B

  • instead, map your fragment onto the canonical live node B'

  • then try to publish the downstream part as

    • B' --e2--> C

if that still makes semantic sense

Why this is the right default

Because the meaning of the fragment is usually not:

“this exact object B must exist”

but rather:

“under edge e1, there is a child state, and from that child there is an e2 successor C”

In concurrent tree structures, object identity is usually incidental. What matters is the canonical node already attached in the tree.

So once B' is the official child under A,e1, all further work should be relative to B', not your discarded private B.

When continuing downstream is correct

Continuing with B' --e2--> C is correct if your downstream information depends only on the semantic state represented by the child under e1, not on the accidental identity of the node object you constructed.

This is the common case.

For example, if:

  • B and B' represent the same proof state / game state / search state

  • and your computation of C was derived from that state

then it is perfectly natural to say:

  • “fine, B' is the canonical version of that state; attach my downstream discovery under B'”

When you should not continue downstream

You should discard the downstream part if it depended on private, noncanonical, or stale assumptions that may not hold for B'.

Examples:

  • B' may not actually represent the same semantic state as your B

  • your computation of C may depend on mutable metadata stored in your private B

  • your downstream expansion may have assumed certain unexpanded children or local caches that are no longer valid once another thread has already modified that area

  • the expansion procedure is nondeterministic, and the downstream result is not guaranteed to be valid for any equivalent replacement of B

In those cases, blindly attaching B' --e2--> C could be wrong.

Then the correct behavior is:

  • reuse B'

  • discard or revalidate the downstream suffix before continuing

So the real rule is

At a conflict on a parent edge:

  1. accept the canonical existing child in the tree

  2. reinterpret the rest of your fragment relative to that canonical child

  3. continue merging only to the extent that the remainder is still semantically valid

That is the general rule.

A useful way to think about it

Your fragment should ideally be treated as a structural recipe, not as a commitment to specific heap objects.

So instead of thinking:

  • A -> my_B -> my_C

think:

  • under A,e1, there should be some canonical child representing state S1

  • under that child, edge e2 should lead to a node representing state S2

Then if another thread already supplied the canonical node for S1, you can still use the rest of the recipe.

Best default policy for a backbone

If you are designing the generic backbone, I would make the default merge rule:

  • conflicts at an edge are resolved by reusing the already-published child

  • downstream publication continues under that reused child

  • but only if the user-specialization says the downstream fragment is still valid under node substitution

Because that last clause is domain-dependent.

Interface consequence

This means the richer expansion interface should probably not model a fragment as “literal node objects wired together” only.

It should also support the notion that:

  • a fragment entry can be re-anchored onto an equivalent canonical node if a race occurs

In other words, publication should be able to say:

  • “your proposed root child lost the race; I substituted the canonical child; now I am continuing merge under that substituted node”

That is a cleaner semantic model than either:

  • always discard downstream or

  • always force downstream through regardless

In your exact notation

If thread 1 conceived B --e2--> C and thread 2 already installed A --e1--> B', the default answer is:

  • try to create B' --e2--> C, not B --e2--> C

  • but do so only if B' is semantically the same anchor that thread 1’s downstream plan was computed from

  • otherwise, stop at B' and discard or recompute the suffix

So the default is “continue under B',” not “give up everything,” but with a semantic-validity guard.

and stops there.

That decision stays entirely in user/application code.

What this buys you

It keeps the backbone conceptually clean.

The backbone only needs to know about:

  • locks

  • atomic edge publication

  • duplicate-edge races

  • snapshots

  • stats / virtual visits / backprop

It does not need to know about:

  • state equivalence

  • whether a stale suffix remains valid

  • whether two node objects represent “the same” theorem state or game state

  • whether downstream expansion data can be transplanted

That is much better.

You basically have two options.

Option 1: keep the backbone primitive narrow

The backbone exposes only safe low-level primitives like:

  • try attach one child

  • get canonical winner on conflict

Then user code orchestrates multi-node expansion on top.

This is the cleanest abstraction boundary.

Option 2: backbone supports multi-node publication, but only with very weak semantics

For example:

  • publish this fragment incrementally

  • if a conflict occurs at some node, stop descending that branch

  • do not attempt semantic substitution or continuation

This can still be generic, because the rule is purely structural.

In other words, the backbone could support “best-effort fragment publication,” but with a simple policy:

descend a branch only so long as each attachment succeeds freshly; on first conflict, truncate that branch.

That rule is semantics-free.

A good generic policy

If you want a richer backbone interface without semantic entanglement, I would recommend exactly that policy:

conflict truncates the affected suffix

Meaning:

  • if A -> e1 conflicts, the backbone reuses the canonical existing child for bookkeeping purposes if needed,

  • but it does not automatically try to attach the descendants that were computed relative to the losing private child,

  • that branch of the fragment simply stops there.

So in your example:

  • thread 1 conceived B --e2--> C

  • but A --e1--> B' already exists

then the generic backbone behavior is:

  • keep B'

  • discard the suffix under private B

  • do not attempt B' --e2--> C

This is conservative, but very clean.

Because it is completely semantics-agnostic.

The backbone never has to answer:

  • is B' “equivalent enough” to B?

  • is C still valid?

  • may the suffix be transplanted?

It just says:

  • “your proposed anchor for this suffix was not the one that became canonical, so I will not continue that suffix.”

That is easy to specify, easy to document, and easy to reason about.

  • let the backbone do semantic substitution logic

  • let user code notice the conflict and explicitly recompute or reuse what it can

Layer 1: backbone primitive

Very small, very clean:

  • attach one child atomically

    • conflict with canonical existing child

That is the only place where the backbone touches shared structure.

Layer 2: optional user-side expansion orchestration

User code may compute a richer expansion artifact, but it is responsible for deciding how to proceed after each attach attempt:

  • if attach succeeds, maybe continue with known descendants

  • if attach conflicts, either:

    • give up suffix

    • recompute from canonical child

    • invoke domain-specific salvage logic

This keeps the backbone generic and keeps sophistication where it belongs: in the specialization.

That is a real interface, and it is generic.

What I would avoid is a stronger interface that promises:

  • subtree merge

  • suffix reuse after conflicts

  • canonical-node substitution with continuation

Those do start dragging semantic assumptions into the backbone.

  • expansion may publish a forest, not just one child

  • rollout work is decoupled from publication work

  • scheduler becomes a first-class abstraction

What is good in it

1. You are making the real unit of work explicit

That is the biggest win.

Your special case is not naturally “expand one edge.” If expansion really discovers a forest, then forcing that into a one-child interface is artificial. Promoting the interface to expand() is the right instinct.

2. Structural publication remains local

Having the backbone attach edges one at a time under one anchor lock at a time is the correct structural discipline. That keeps the concurrency story clean.

3. Rollout becomes a schedulable resource

That is also good. In many applications, expansion and rollout have very different cost profiles. Making the scheduler explicit lets you separate:

  • tree mutation work

  • expensive evaluation work

That is a real architectural improvement.

4. “Conflict resolver” is cleaner than semantic magic inside the backbone

This is better than the backbone silently trying to reinterpret stale suffixes. At least now the customization point is explicit.

The first big issue: the proposed list[tuple[NodeT, EdgeT, NodeT, bool]] is too raw

I do not think that should be the public shape.

It is expressive enough, but it is under-structured in exactly the places where future developers will get hurt.

A flat list of quadruples leaves too many semantics implicit:

  • What ordering constraints exist between entries?

  • Are later entries allowed to reference children introduced by earlier entries?

  • Must the list be topologically ordered?

  • Can multiple entries mark bool=True?

  • Must there be at least one such node?

  • May a node with bool=True also have descendants in later entries?

  • What exactly does it mean for a node to be “rolled out directly” if it also gets descendants in the same batch?

You can define all of that in prose, but that is usually a sign the type is not carrying enough structure.

I would instead make the return value a named object, conceptually something like an ExpansionBatch or ExpansionForest, containing:

  • edge insertions

  • rollout targets

  • maybe optional metadata about ordering/topology

The list-of-quadruples can be the internal normalized form if you want, but I would not make it the main abstraction boundary.

Second big issue: bool is too weak

A boolean “roll this node out” flag is underspecified.

There are at least three distinct statuses a newly created node might have:

  • materialize only

  • enqueue rollout

  • already terminal / already evaluated / no rollout needed

Those are not the same thing.

Also, what if multiple nodes are marked for rollout? Then your design is no longer “one simulation per expansion.” That may be fine, but it is a substantial semantic shift. A boolean hides that shift instead of surfacing it.

I would replace the boolean with an explicit rollout disposition, conceptually an enum-like choice:

  • no rollout

  • enqueue rollout

  • terminal value available now

  • maybe reuse supplied evaluation

Even if you only implement two cases initially, I would design for that vocabulary.

Third big issue: this is no longer classical MCTS in an innocuous sense

This is not just a harmless generalization. It changes the accounting model in a deep way.

Classical MCTS iteration is roughly:

  • choose one path

  • expand one frontier point

  • evaluate one leaf

  • backprop one result

Your design allows:

  • one selection/expansion step

  • publication of many new nodes

  • enqueue of multiple rollouts

  • later asynchronous completions

  • multiple future backprops stemming from one expansion batch

That is not “classical MCTS but with better plumbing.” It is a broader search/evaluation framework inspired by MCTS.

That is okay. But I would not try too hard to pretend full equivalence. The default recoverability story is useful, but conceptually this is now a more general backbone.

Fourth big issue: virtual visits on “nodes pending rollout” is not enough by itself

This is the most important technical concern.

You said: use virtual visits on nodes pending a rollout. That is directionally right, but the precise object that is pending is not merely a node. It is a work item with a future result.

You now need to distinguish at least four things:

  • node has been materialized

  • node has one or more rollouts enqueued

  • some of those rollouts are still pending

  • some have completed and contributed backed-up value

If a node can receive multiple asynchronously scheduled rollouts, then simple in_flight: int still works as a count, but you also need to define:

  • when exactly that count increments

  • when it decrements

  • whether duplicate rollout submissions to the same node are allowed

  • whether selection should see all pending rollouts equally

  • whether a node can be expanded again while rollouts are still pending for it

Those rules matter a lot.

So I would not frame it as “virtual visits on nodes pending a rollout.” I would frame it as:

  • the scheduler creates evaluation tickets

  • those tickets induce virtual visits along an associated backprop path

  • completion of a ticket resolves those virtual visits and contributes a value

That is a clearer mental model.

Fifth big issue: what path receives the virtual visits?

In ordinary parallel MCTS, this is easy: the selected root-to-leaf path.

In your design, if expansion returns a forest and several newly added nodes are marked for rollout, then there may be several distinct pending backprop paths created by one expansion batch.

That is fine, but it must be explicit.

Suppose expansion at A publishes:

  • A -> B1

  • A -> B2 and both B1 and B2 are marked for rollout.

Then are you creating:

  • two pending evaluation tickets

  • one for path root -> ... -> A -> B1

  • one for path root -> ... -> A -> B2

If yes, then both of those paths get virtual visits. That is a substantial departure from one-simulation-at-a-time semantics, but it is coherent.

You just need to define it that way clearly.

Sixth big issue: the forest constraint needs to be stronger than “must be a forest”

“Forest with roots already in the MCTS tree” is not yet enough.

You also need constraints like:

  • the returned edges must be acyclic

  • if an entry references an anchor not already in the tree, that anchor must have appeared earlier in the same batch and must have been successfully published, or else that branch is invalid

  • the batch should probably be topologically ordered

  • multiple entries for the same (anchor, edge) in one batch should be forbidden

  • the same child object should not appear under two different parents unless you explicitly support DAG/transposition semantics

Without those rules, debugging downstream misuse will be miserable.

So yes, “forest” is the right intuition, but you need a much sharper contract.

Seventh big issue: the conflict resolver is okay, but I would narrow its role

A default resolver that discards the incoming child is fine.

An overridable resolver is also fine, but I would be careful not to let it turn into a general semantic merge engine inside the backbone.

The clean role for conflict_resolver(anchor, edge, incoming_child) is something like:

Eighth big issue: your scheduler/barrier design is a reasonable default, but it is probably too global as the primary scheduler model

The “phase 1 select/expand, phase 2 rollout, synchronized by barrier” design is a decent baseline, especially if you want deterministic-ish bulk behavior.

But it imposes a very specific worldview:

  • the whole system alternates between tree-growth mode and evaluation mode

That is not always what you want. In many workloads, you want overlap:

  • some workers publishing new structure

  • others consuming rollout tasks already available

So I agree with your instinct to make scheduler user-definable. I just would not present the barrier-phase scheduler as anything more than a simple reference implementation.

It is fine as a default, but not as the conceptual center of the architecture.

Ninth big issue: “#workers can be decoupled from concurrency” is correct, but define concurrency carefully

You need to distinguish:

  • number of worker threads

  • max number of concurrent expansion tasks

  • max number of concurrent rollout tasks

  • max number of in-flight evaluation tickets

Otherwise “concurrency” becomes too vague.

This matters because virtual-visit pressure and memory pressure are driven more by in-flight evaluation tickets than by raw thread count.

So I would have the scheduler abstraction talk in terms of queues/capacities, not just workers.

Tenth big issue: expansion result should probably not directly contain NodeT children in the most abstract interface

This is less critical, but worth considering.

Returning concrete child node objects is simple, but it means expansion eagerly constructs all child objects even if publication later conflicts. That may be acceptable. But if your “free descendants” are numerous and expensive, the waste may become large.

  • keep concrete NodeT in the interface because simplicity matters most

  • or allow child specs/factories as an advanced option

I would probably start with concrete nodes, but I would design the docs so that adding a lazy/spec variant later would not be an API betrayal.

What I would change in your proposal

1. Replace the flat tuple list with a structured batch object

Not for performance — for semantics.

  • attachments

  • rollout targets

  • maybe topological order

2. Replace bool with explicit evaluation disposition

This avoids ambiguity and future-proofs the interface.

3. Model rollout submissions as first-class work items

Each such item should carry:

  • evaluation target node

  • associated ancestor path for virtual visits/backprop

  • maybe a generation/token so completion can be matched safely

4. Make branch conflict semantics conservative by default

On conflict at one attachment:

  • keep canonical existing edge

  • truncate only that suffix by default

  • do not automatically transplant descendants

That matches your desired cleanliness.

5. Require the expansion batch to be topologically ordered

That will simplify implementation and reduce ambiguity a lot.

6. Separate “materialization batch” from “evaluation requests”

Even if produced by one expand() call, those are conceptually different outputs.

That will make the scheduler cleaner too.

The version of your design I would endorse

Conceptually:

  • expand(anchor) returns an ExpansionBatch

  • the batch contains:

    • a topologically ordered forest of proposed edge insertions

    • a set of evaluation requests for some newly materialized or existing nodes

  • the backbone publishes insertions one attachment at a time under local locks

  • local conflicts are resolved conservatively

  • successfully accepted evaluation requests become scheduler-managed work items

  • each work item installs virtual visits on its associated backprop path until completion

  • rollout workers resolve work items asynchronously

  • completion performs ordinary locked backprop

That is coherent.

What I think the real remaining design questions are

The important questions are now not about bool or node factories. They are these:

1. What is the exact contract on the returned list order?

This matters a lot.

Since your return value is a flat list of (anchor, edge, child, should_rollout), you need a clear rule for when an anchor in a later tuple is allowed to be a node that only appeared as a child in an earlier tuple.

I think the clean contract is:

  • the list must be in topological order

  • every anchor must either:

    • already be present in the live MCTS tree before expand() began, or

    • have appeared earlier in the same returned list as a child

Without that, the backbone cannot process the list one item at a time in a straightforward way.

2. What happens if an earlier attachment in the list conflicts?

This is the biggest one.

Suppose expand() returns:

  • (A, e1, B, False)

  • (B, e2, C, True)

but when the backbone tries the first tuple, A --e1--> B' already exists.

Then the second tuple is now anchored at a node B that was never attached.

You already gave the philosophical answer earlier: the backbone should stay conservative and not try semantic substitution on its own.

So I think the contract should be:

  • if a tuple’s attachment conflicts, the incoming child is passed to conflict_resolver

  • if the incoming child does not become the canonical attached child, then any later tuples anchored at that losing child object are simply skipped

That is a very clean rule.

It means the returned list is interpreted relative to object identity, not semantic equivalence. That keeps the backbone simple.

3. How should virtual visits work when multiple True nodes are returned?

This is the key algorithmic question, but it is still manageable.

I think the natural rule is:

  • every tuple with should_rollout=True that is successfully materialized spawns an independent pending rollout task

  • each such task installs virtual visits along its own ancestor chain to root

  • when the rollout finishes, those virtual visits are resolved into real stats by backprop

This does generalize the single-rollout-path model cleanly.

The one thing I would document explicitly is that this may create several simultaneous in-flight rollout paths from one expand() call. That is not a bug; it is the intended generalized semantics.

4. Can a node be selected/expanded while it already has pending rollout work?

I think the answer should be yes.

Otherwise the scheduler/backbone becomes much more rigid than necessary.

Selection should just “see” pending work through virtual visits, which naturally discourages repeated hammering without banning it outright.

5. Is the returned structure allowed to contain multiple roots?

Yes, and that seems fine. But then your “forest rooted in existing tree nodes” phrasing should probably be sharpened to:

  • the returned list may describe several branches

  • branches may share already-existing anchors

  • newly proposed children in the same batch must not have two parents

That last condition matters.

I think the design is good, provided you explicitly specify these backbone rules:

Expansion return contract

expand(node) returns a flat list of (anchor, edge, child, should_rollout) such that:

  • the list is topologically ordered

  • each anchor is either already in the tree or was introduced earlier in the same list

  • the proposed new edges form a forest

  • no proposed child has two parents

  • no duplicate (anchor, edge) appears within one returned list

Publication semantics

The backbone processes the list in order, one tuple at a time:

  • attach under the anchor with only that anchor’s lock held

  • if the edge is free, attach the child

  • if the edge conflicts, invoke conflict_resolver(anchor, edge, incoming_child)

  • if the incoming child is not the canonical attached child after resolution, then later list entries anchored at that incoming child are skipped

That last clause is the key conservative semantics.

Rollout semantics

For every successfully materialized tuple with should_rollout=True:

  • enqueue one rollout task

  • install virtual visits for that task’s path

  • resolve those virtual visits when rollout/backprop completes

Scheduler semantics

MCTSWorkerScheduler is responsible for:

  • dispatching select/expand jobs

  • dispatching rollout jobs

  • respecting whatever concurrency policy it implements

And your default barrier-phase scheduler is a reasonable baseline, but just a baseline.

all semantic cleverness stays in user code, while the backbone still gives first-class support for multi-node expansion and async rollout.
